## Installation

```bash
npm install --save react-forward-head-posture
```

### Usage

```jsx
import ForwardHeadPosture from "react-forward-head-posture"

const { log } = console

export default function App() {
  return <ForwardHeadPosture onEstimate={log} />
}
```
